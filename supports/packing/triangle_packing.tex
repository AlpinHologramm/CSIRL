%!TEX encoding = UTF-8 Unicode

\begin{tikzpicture}

% define base units ------------------------------------------------------------
	\def\delta{1/10}

% draw grid --------------------------------------------------------------------
	\draw[step=1, gray, very thin] (-1,-1) grid (9,7);

% draw rectangles --------------------------------------------------------------
	\def\rect{
%		coord_x / coord_y / size_x /size_y
		0 / 0 / 3 / 3,
		0 / 3 / 4 / 1,
		0 / 4 / 3 / 2,
		3 / 0 / 1 / 3,
		3 / 5 / 3 / 1,
		4 / 0 / 4 / 1,
		4 / 1 / 4 / 2,
		4 / 3 / 2 / 2,
		6 / 3 / 2 / 3
	}
	\foreach \x / \y / \sx / \sy in \rect
	{
		\draw[thick] (\x,\y) rectangle +(\sx,\sy);
		\fill[shift={(\x+1.2\delta,\y+1.2\delta)}] (0:-0.6\delta) -- (120:-0.6\delta) -- (-120:-0.6\delta) -- cycle;
	}

% draw holes -------------------------------------------------------------------
	\def\hole{
%		coord_x / coord_y / size_x /size_y
		3 / 4 / 1 / 1
	}
	\foreach \x / \y / \sx / \sy in \hole
	{
		\fill[pattern=north east lines] (\x,\y) rectangle +(\sx,\sy);
	}

\end{tikzpicture}
