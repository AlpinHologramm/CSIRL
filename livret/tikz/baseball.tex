\newcommand{\base}[5]{
		\draw [shift={(#5)},fill=#1,thick] (0,0) circle (1cm);
		\draw [shift={(#5)},fill=#2,thick,rotate=#4] (-0.25,-0.5) rectangle (0.25,0);
		\draw [shift={(#5)},fill=#3,thick,rotate=#4] (-0.25,0) rectangle (0.25,0.5);	
}

\newcommand{\etat}[8]{
\begin{tikzpicture}[scale=0.5]
	\draw [dotted] (0,0) circle (2cm);
	\base{red}{#1}{#2}{-45}{(45:2cm)}
	\base{green}{#3}{#4}{45}{(45+90:2cm)}
	\base{blue}{#5}{#6}{45+90}{(45+180:2cm)}
	\base{yellow}{#7}{#8}{45+180}{(45+270:2cm)}
\end{tikzpicture}
}
